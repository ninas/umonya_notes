5. --- zip_lists.py ---
names = []
ages = []

name = raw_input('Enter a name: ')
while name != '':
	names.append(name)
	age = int(raw_input('Enter age for '+name+': '))
	ages.append(age)
	name = raw_input('Enter a name: ')

result = []
for i in range(len(names)):
	result.append((names[i], ages[i]))

print result
--- end zip_lists.py ---
