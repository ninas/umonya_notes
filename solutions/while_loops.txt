2. A line that increments i by 1 needs to be put at the end of, but inside of
the while loop.

3. --- smallest.py ---
smallest = "strings are always larger than integers"
print "Enter a sequence of numbers:"
line = raw_input("> ")
while line != "":
	if int(line) < smallest:
		smallest = int(line)
	line = raw_input("> ")
print smallest

4. --- average.py ---
print "Enter a sequence of numbers:"
total = 0
count = 0
line = raw_input("> ")
while line != "":
	total += int(line)
	count += 1
	line = raw_input("> ")
print float(total)/count
--- end average.py ---
